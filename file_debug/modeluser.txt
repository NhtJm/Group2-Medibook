<?php
require_once __DIR__ . '/../app/model/UserModel.php';

$userModel = new UserModel();

// =============================
// TEST 1: Thêm user mới
// =============================
echo "<h3>== Test 1: Add User ==</h3>";
$userRes = $userModel->addUser("john_doe_test11", "123456", "John Doe");
print_r($userRes);
$user_id = $userRes['user_id'] ?? null;

if (!$user_id) {
    die("Cannot create user. Stop testing.");
}

// =============================
// TEST 2: Lấy thông tin user theo ID
// =============================
echo "<h3>== Test 2: Get User By ID ==</h3>";
$user = $userModel->getUserById($user_id);
print_r($user);

// =============================
// TEST 3: Lấy thông tin user theo username
// =============================
echo "<h3>== Test 3: Get User By Username ==</h3>";
$userByName = $userModel->getUserByUsername("john_doe_test");
print_r($userByName);

// =============================
// TEST 4: Cập nhật tên user
// =============================
echo "<h3>== Test 4: Update User Name ==</h3>";
$updateRes = $userModel->updateUser($user_id, ["name" => "John Updated"]);
print_r($updateRes);

// =============================
// TEST 5: Thêm số điện thoại
// =============================
echo "<h3>== Test 5: Add Phone ==</h3>";
$phoneRes = $userModel->addPhone($user_id, "0123456789");
print_r($phoneRes);

$phones = $userModel->getPhonesByUser($user_id);
echo "Current Phones: ";
print_r($phones);

// =============================
// TEST 6: Cập nhật số điện thoại
// =============================
echo "<h3>== Test 6: Update Phone ==</h3>";
$updatePhone = $userModel->updatePhone($user_id, "0123456789", "0987654321");
print_r($updatePhone);

$phones = $userModel->getPhonesByUser($user_id);
echo "Updated Phones: ";
print_r($phones);

// =============================
// TEST 7: Xóa số điện thoại
// =============================
echo "<h3>== Test 7: Delete Phone ==</h3>";
$deletedPhone = $userModel->deletePhone($user_id, "0987654321");
print_r($deletedPhone);

$phones = $userModel->getPhonesByUser($user_id);
echo "Phones after deletion: ";
print_r($phones);

// =============================
// TEST 8: Thêm patient
// =============================
echo "<h3>== Test 8: Add Patient ==</h3>";
$patientRes = $userModel->addPatient($user_id, "2000-01-01", "VNPay");
print_r($patientRes);
$patient_id = $patientRes['patient_id'] ?? null;

// =============================
// TEST 9: Cập nhật patient
// =============================
echo "<h3>== Test 9: Update Patient ==</h3>";
if ($patient_id) {
    $updatePatient = $userModel->updatePatient($patient_id, ["status" => "inactive"]);
    print_r($updatePatient);
    $patient = $userModel->getPatientByUser($user_id);
    print_r($patient);
}

// =============================
// TEST 10: Thêm staff
// =============================
echo "<h3>== Test 10: Add Staff ==</h3>";
$staffRes = $userModel->addStaff($user_id, "support");
print_r($staffRes);
$staff_id = $staffRes['staff_id'] ?? null;

// =============================
// TEST 11: Cập nhật staff
// =============================
echo "<h3>== Test 11: Update Staff ==</h3>";
if ($staff_id) {
    $updateStaff = $userModel->updateStaff($staff_id, ["status" => "online"]);
    print_r($updateStaff);
    $staff = $userModel->getStaffByUser($user_id);
    print_r($staff);
}

// =============================
// TEST 12: Xóa staff
// =============================
echo "<h3>== Test 12: Delete Staff ==</h3>";
if ($staff_id) {
    $deletedStaff = $userModel->deleteStaff($staff_id);
    print_r($deletedStaff);
}

// =============================
// TEST 13: Xóa patient
// =============================
echo "<h3>== Test 13: Delete Patient ==</h3>";
if ($patient_id) {
    $deletedPatient = $userModel->deletePatient($patient_id);
    print_r($deletedPatient);
}

// =============================
// TEST 14: Xóa user
// =============================
echo "<h3>== Test 14: Delete User ==</h3>";
if ($user_id) {
    $deletedUser = $userModel->deleteUser($user_id);
    print_r($deletedUser);
}

echo "<h3>All tests completed.</h3>";
?>
