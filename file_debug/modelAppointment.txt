<?php
require_once __DIR__ . '/../app/model/AppointmentModel.php';

$appointmentModel = new AppointmentModel();

echo "<h2>=== TEST APPOINTMENT MODEL ===</h2>";

// NOTE: Bạn cần chỉnh các ID mẫu dưới đây cho đúng với dữ liệu hiện có trong database
$patient_id = 1;   // ID của 1 patient hợp lệ
$slot_id    = 29;   // ID của 1 slot hợp lệ (trạng thái "available")

// =============================
// TEST 1: Thêm cuộc hẹn mới
// =============================
echo "<h3>== Test 1: Add Appointment ==</h3>";
$resAdd = $appointmentModel->addAppointment($patient_id, $slot_id, 'booked');
print_r($resAdd);

$appointment_id = $resAdd['appointment_id'] ?? null;
if (!$appointment_id) {
    die("❌ Cannot create appointment — check patient_id / slot_id validity.<br>");
}

// =============================
// TEST 2: Lấy tất cả appointments
// =============================
echo "<h3>== Test 2: Get All Appointments ==</h3>";
$all = $appointmentModel->getAllAppointments();
print_r($all);

// =============================
// TEST 3: Lấy appointment theo ID
// =============================
echo "<h3>== Test 3: Get Appointment By ID ==</h3>";
$appt = $appointmentModel->getAppointmentById($appointment_id);
print_r($appt);

// =============================
// TEST 4: Lấy appointment theo bệnh nhân
// =============================
echo "<h3>== Test 4: Get Appointments By Patient ==</h3>";
$byPatient = $appointmentModel->getAppointmentsByPatient($patient_id);
print_r($byPatient);

// =============================
// TEST 5: Lấy appointment theo bác sĩ
// =============================
echo "<h3>== Test 5: Get Appointments By Doctor ==</h3>";
$doctor_id = 1; // ID của bác sĩ trong DB
$byDoctor = $appointmentModel->getAppointmentsByDoctor($doctor_id);
print_r($byDoctor);

// =============================
// TEST 6: Cập nhật trạng thái appointment
// =============================
echo "<h3>== Test 6: Update Appointment Status ==</h3>";
$update = $appointmentModel->updateAppointmentStatus($appointment_id, 'completed');
print_r($update);

// =============================
// TEST 7: Thử cập nhật trạng thái invalid
// =============================
echo "<h3>== Test 7: Invalid Status Update ==</h3>";
$invalid = $appointmentModel->updateAppointmentStatus($appointment_id, 'unknown_status');
print_r($invalid);

// =============================
// TEST 8: Xóa appointment (nếu chưa completed)
// =============================
echo "<h3>== Test 8: Delete Appointment ==</h3>";
$delete = $appointmentModel->deleteAppointment($appointment_id);
print_r($delete);

// =============================
// TEST 9: Kiểm tra lại danh sách sau khi xóa
// =============================
echo "<h3>== Test 9: Get All After Delete ==</h3>";
$afterDelete = $appointmentModel->getAllAppointments();
print_r($afterDelete);

echo "<h3>All Appointment tests completed ✅</h3>";
?>
